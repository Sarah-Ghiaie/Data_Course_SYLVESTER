3. Write the code you would use to model the data found in “/Data/non_linear_relationship.csv” with a linear model (there are a few ways of doing this).

#load data
nonlin <- read_csv("Data/non_linear_relationship.csv")
#simple straight-line model
mod_simple <- lm(response ~ predictor, data = nonlin)
#linear model with a quadratic term
mod_quad <- lm(response ~ predictor + I(predictor^2), data = nonlin)
#polynomial form using poly()
mod_poly <- lm(response ~ poly(predictor, 2), data = nonlin)
